function shuffle(arr) {
  const a = [...arr];
  for (let i = a.length - 1; i > 0; i--) {
    const j = Math.floor(Math.random() * (i + 1));
    [a[i], a[j]] = [a[j], a[i]];
  }
  return a;
}
 
function getChoices(correct) {
  const pool = words.map(w => w[1]).filter(t => t !== correct);
  const picks = shuffle(pool).slice(0, 3);
  return shuffle([correct, ...picks]);
}

function checkAnswer(selected, btn, allBtns) {
  const correct = currentWord()[1];
  allBtns.forEach(b => b.disabled = true);
 
  if (selected === correct) {
    btn.classList.add('correct');
    score++; streak++; answered++;
    showToast('✅ נכון!', 'success');
  } else {
    btn.classList.add('wrong');
    // reveal correct
    allBtns.forEach(b => { if (b.dataset.val === correct) b.classList.add('correct'); });
    streak = 0; wrongCount++; answered++;
    const word = currentWord()[0];
    const entry = word + '|' + correct;
    if (!wrongWords.includes(entry)) wrongWords.push(entry);
    renderWrongList();
    showToast('❌ ' + correct, 'error');
  }
  updateStats();
}
 
function renderWrongList() {
  const div = document.getElementById('wrongList');
  const badge = document.getElementById('wrongBadge');
  badge.textContent = wrongWords.length;
  if (wrongWords.length === 0) {
    div.innerHTML = '<div class="wrong-empty">עדיין אין מילים שגויות 🌱</div>';
  } else {
    div.innerHTML = wrongWords.map(entry => {
      const [en, he] = entry.split('|');
      return `<div class="wrong-item"><span class="wrong-en">${en}</span><span class="wrong-he">${he}</span></div>`;
    }).join('');
  }
}
 
function renderQuiz() {
  updateStats();
  const [word, translation] = currentWord();
 
  document.getElementById('wordDisplay').textContent = word;
  const reveal = document.getElementById('answerReveal');
  reveal.textContent = translation;
  reveal.classList.remove('visible');
 
  const grid = document.getElementById('choicesGrid');
  grid.innerHTML = '';
  const choices = getChoices(translation);
  choices.forEach(c => {
    const btn = document.createElement('button');
    btn.className = 'choice-btn';
    btn.textContent = c;
    btn.dataset.val = c;
    btn.onclick = () => checkAnswer(c, btn, [...grid.querySelectorAll('.choice-btn')]);
    grid.appendChild(btn);
  });
 
  document.getElementById('modeIndicator').textContent = isShuffled ? '🔀 מצב ערבוב' : '';
}
 
function navigate(dir) {
  currentPos = (currentPos + dir + words.length) % words.length;
  renderQuiz();
}
 
// Event listeners
document.getElementById('speakBtn').onclick = () => speakWord(currentWord()[0]);
document.getElementById('revealBtn').onclick = () => {
  document.getElementById('answerReveal').classList.add('visible');
};
document.getElementById('prevBtn').onclick = () => navigate(-1);
document.getElementById('nextBtn').onclick = () => navigate(1);
document.getElementById('shuffleBtn').onclick = () => {
  isShuffled = !isShuffled;
  wordOrder = isShuffled ? shuffle(words.map((_, i) => i)) : words.map((_, i) => i);
  currentPos = 0;
  renderQuiz();
};
 
// Wrong list toggle
document.getElementById('wrongToggle').onclick = () => {
  wrongOpen = !wrongOpen;
  document.getElementById('wrongList').classList.toggle('open', wrongOpen);
  document.getElementById('wrongChevron').classList.toggle('open', wrongOpen);
};
 
// Keyboard navigation
document.addEventListener('keydown', e => {
  if (e.key === 'ArrowLeft') navigate(1);
  if (e.key === 'ArrowRight') navigate(-1);
  if (e.key === ' ') { e.preventDefault(); speakWord(currentWord()[0]); }
  const n = parseInt(e.key);
  if (n >= 1 && n <= 4) {
    const btns = document.querySelectorAll('.choice-btn:not(:disabled)');
    if (btns[n-1]) btns[n-1].click();
  }
});
 
renderWrongList();
renderQuiz();
